# 去这里下载最新的jar包
# [All releases - The Go Programming Language](https://go.dev/dl/)
# 拿 go1.20.5.linux-amd64.tar.gz 举例

# 解压到/usr/local/
tar -C /usr/local/ -xzf go1.20.5.linux-amd64.tar.gz

# 文件内容

# 加入 ~/.bashrc,每次生效
export PATH=$PATH:/usr/local/go/bin
# 此次终端生效
source ~/.bashrc

# 验证效果
