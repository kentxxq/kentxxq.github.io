TYPE=Ethernet
BOOTPROTO=static
DEFROUTE=yes

PEERROUTES=yes
IPV4_FAILURE_FATAL=no
IPV6INIT=yes
IPV6_AUTOCONF=yes
IPV6_DEFROUTE=yes
IPV6_PEERDNS=yes
IPV6_PEERROUTES=yes
IPV6_FAILURE_FATAL=no
IPV6_ADDR_GEN_MODE=stable-privacy

UUID=774dda00-53bc-49d3-9b7a-ea38e397ba84
DEVICE=enp0s3
ONBOOT=yes
IPADDR=192.168.0.222
NETMASK=255.255.255.0
GATEWAY=192.168.0.1
