第一步–创建virtual box虚拟机

第二步–默认安装、最小化安装

使用的iso为CentOS-7-x86_64-Everything-1611.iso centos官网下载
配置root用户密码、新建用户等

第三步–网卡配置

ssh root@192.168.0.222
cd /etc/sysconfig/network-scripts

ifcfg-enp0s3 使用的网卡
ifcfg-lo 回环
配置ifcfg-enp0s3,其中UUID无需更改

[root@centos1 network-scripts]# cat ifcfg-enp0s3 
TYPE=Ethernet
BOOTPROTO=static
DEFROUTE=yes
PEERDNS=yes
PEERROUTES=yes
IPV4_FAILURE_FATAL=no
IPV6INIT=yes
IPV6_AUTOCONF=yes
IPV6_DEFROUTE=yes
IPV6_PEERDNS=yes
IPV6_PEERROUTES=yes
IPV6_FAILURE_FATAL=no
IPV6_ADDR_GEN_MODE=stable-privacy
NAME=enp0s3
UUID=774dda00-53bc-49d3-9b7a-ea38e397ba84
DEVICE=enp0s3
ONBOOT=yes
IPADDR=192.168.0.222
NETMASK=255.255.255.0
GATEWAY=192.168.0.1

重启服务即可正常使用

[root@centos1 network-scripts]# service network restart
Restarting network (via systemctl):                        [  确定  ]
[root@centos1 network-scripts]# ping baidu.com
PING baidu.com (111.13.101.208) 56(84) bytes of data.
64 bytes from localhost (111.13.101.208): icmp_seq=1 ttl=54 time=38.7 ms
